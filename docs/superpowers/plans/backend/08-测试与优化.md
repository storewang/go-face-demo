```python
import pytest
import numpy as np
from app.utils.face_utils import FaceUtils

class TestFaceUtils:
    
    def test_load_image_from_bytes(self):
        """测试从字节加载图像"""
        # 创建测试图像 (JPEG)
        import cv2
        test_img = np.zeros((100, 100, 3), dtype=np.uint8)
        _, encoded = cv2.imencode('.jpg', test_img)
        image_bytes = encoded.tobytes()
        
        result = FaceUtils.load_image_from_bytes(image_bytes)
        
        assert result.shape == (100, 100, 3)
        assert result.dtype == np.uint8
    
    def test_get_face_quality_poor(self):
        """测试人脸质量评估 - 差"""
        # 小人脸
        small_face = np.random.randint(0, 255, (50, 50, 3), dtype=np.uint8)
        location = (0, 50, 50, 0)
        
        quality = FaceUtils.get_face_quality(small_face, location)
        
        assert quality == "poor"
    
    def test_get_face_quality_good(self):
        """测试人脸质量评估 - 好"""
        # 大人脸，亮度适中
        good_face = np.full((200, 200, 3), 128, dtype=np.uint8)
        location = (0, 200, 200, 0)
        
        quality = FaceUtils.get_face_quality(good_face, location)
        
        assert quality in ["good", "medium"]  # 清晰度可能影响结果
```

## 六、日志配置

### 6.1 日志配置 (app/logging_config.py)

```python
import logging
from logging.handlers import RotatingFileHandler
from pathlib import Path

def setup_logging():
    """配置日志"""
    log_dir = Path("logs")
    log_dir.mkdir(exist_ok=True)
    
    # 格式
    formatter = logging.Formatter(
        '%(asctime)s - %(name)s - %(levelname)s - %(message)s'
    )
    
    # 文件处理器
    file_handler = RotatingFileHandler(
        log_dir / "app.log",
        maxBytes=10*1024*1024,  # 10MB
        backupCount=5
    )
    file_handler.setFormatter(formatter)
    
    # 控制台处理器
    console_handler = logging.StreamHandler()
    console_handler.setFormatter(formatter)
    
    # 根日志器
    logger = logging.getLogger()
    logger.setLevel(logging.INFO)
    logger.addHandler(file_handler)
    logger.addHandler(console_handler)
    
    return logger
```
