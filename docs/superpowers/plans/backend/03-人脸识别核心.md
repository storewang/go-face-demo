## 一、目标

实现人脸识别核心服务，包括人脸检测、质量评估、特征提取、1:N比对功能。

## 二、技术方案

### 2.1 核心库

- **face_recognition**: 基于 dlib 的人脸识别库
  - `face_locations()`: 检测人脸位置 (HOG 或 CNN)
  - `face_encodings()`: 提取128维特征向量
  - `face_distance()`: 计算特征向量距离

### 2.2 关键算法

```python
# 人脸检测
locations = face_recognition.face_locations(image, model="hog")

# 特征提取
encodings = face_recognition.face_encodings(image, locations)

# 1:N 比对
distances = face_recognition.face_distance(known_encodings, unknown_encoding)
# distance 越小越相似，通常 < 0.6 认为匹配

# 转换为置信度 (0-1)
confidence = 1 - min(distances)
```

### 3.2 人脸工具类 (app/utils/face_utils.py)

```python
import cv2
import numpy as np
import face_recognition
from typing import Tuple, List, Optional

class FaceUtils:
    """人脸处理工具类"""
    
    @staticmethod
    def load_image_from_bytes(image_bytes: bytes) -> np.ndarray:
        """从字节数据加载图像 (RGB格式)"""
        nparr = np.frombuffer(image_bytes, np.uint8)
        image = cv2.imdecode(nparr, cv2.IMREAD_COLOR)
        # OpenCV 默认 BGR，转为 RGB
        return cv2.cvtColor(image, cv2.COLOR_BGR2RGB)
    
    @staticmethod
    def load_image_from_file(file_path: str) -> np.ndarray:
        """从文件加载图像"""
        image = face_recognition.load_image_file(file_path)
        return image
    
    @staticmethod
    def get_face_quality(image: np.ndarray, location: Tuple) -> str:
        """
        评估人脸质量
        
        Args:
            image: RGB图像
            location: (top, right, bottom, left)
        
        Returns:
            "good" | "medium" | "poor"
        """
        top, right, bottom, left = location
        face_height = bottom - top
        face_width = right - left
        
        # 尺寸检查
        if face_width < 80 or face_height < 80:
            return "poor"
        if face_width < 120 or face_height < 120:
            return "medium"
        
        # 亮度检查
        face_region = image[top:bottom, left:right]
        brightness = np.mean(face_region)
        if brightness < 50 or brightness > 200:
            return "poor"
        if brightness < 80 or brightness > 180:
            return "medium"
        
        # 清晰度检查（拉普拉斯方差）
        gray = cv2.cvtColor(face_region, cv2.COLOR_RGB2GRAY)
        clarity = cv2.Laplacian(gray, cv2.CV_64F).var()
        if clarity < 50:
            return "poor"
        if clarity < 100:
            return "medium"
        
        return "good"
    
    @staticmethod
    def draw_face_box(image: np.ndarray, location: Tuple, label: str = "") -> np.ndarray:
        """在图像上绘制人脸框"""
        top, right, bottom, left = location
        
        # 转为 BGR (OpenCV格式)
        image_bgr = cv2.cvtColor(image, cv2.COLOR_RGB2BGR)
        
        # 绘制矩形
        cv2.rectangle(image_bgr, (left, top), (right, bottom), (0, 255, 0), 2)
        
        # 绘制标签
        if label:
            cv2.rectangle(image_bgr, (left, bottom - 25), (right, bottom), (0, 255, 0), cv2.FILLED)
            cv2.putText(image_bgr, label, (left + 6, bottom - 6), 
                       cv2.FONT_HERSHEY_SIMPLEX, 0.5, (255, 255, 255), 1)
        
        return image_bgr
```

### 3.3 人脸服务 (app/services/face_service.py)

```python
import os
import numpy as np
import face_recognition
from typing import List, Tuple, Optional, Dict
from pathlib import Path
from sqlalchemy.orm import Session

from app.config import settings
from app.models import User
from app.utils.face_utils import FaceUtils

class FaceService:
    """人脸识别服务"""
    
    def __init__(self):
        self.known_encodings: List[np.ndarray] = []
        self.known_users: List[Dict] = []
        self.threshold = settings.FACE_THRESHOLD
        self._load_known_faces()
    
    def _load_known_faces(self):
        """启动时加载所有已知人脸特征"""
        from app.database import SessionLocal
        db = SessionLocal()
        
        try:
            users = db.query(User).filter(
                User.status == 1,
                User.face_encoding_path.isnot(None)
            ).all()
            
            for user in users:
                if os.path.exists(user.face_encoding_path):
                    encoding = np.load(user.face_encoding_path)
                    self.known_encodings.append(encoding)
                    self.known_users.append({
                        "id": user.id,
                        "employee_id": user.employee_id,
                        "name": user.name,
                        "department": user.department
                    })
            
            print(f"✅ Loaded {len(self.known_encodings)} known faces")
        finally:
            db.close()
    
    def reload_faces(self, db: Session):
        """重新加载人脸特征（注册/删除用户后调用）"""
        self.known_encodings = []
        self.known_users = []
        self._load_known_faces_from_db(db)
    
    def _load_known_faces_from_db(self, db: Session):
        """从数据库加载人脸特征"""
        users = db.query(User).filter(
            User.status == 1,
            User.face_encoding_path.isnot(None)
        ).all()
        
        for user in users:
            if os.path.exists(user.face_encoding_path):
                encoding = np.load(user.face_encoding_path)
                self.known_encodings.append(encoding)
                self.known_users.append({
                    "id": user.id,
                    "employee_id": user.employee_id,
                    "name": user.name,
                    "department": user.department
                })
    
    def detect_faces(self, image: np.ndarray) -> List[Dict]:
        """
        检测图像中的人脸
        
        Args:
            image: RGB图像数组
        
        Returns:
            [{"box": (top, right, bottom, left), "encoding": np.ndarray, "quality": str}]
        """
        # 使用 HOG 模型（更快）
        locations = face_recognition.face_locations(image, model="hog")
        
        if not locations:
            return []
        
        # 提取特征
        encodings = face_recognition.face_encodings(image, locations)
        
        results = []
        for location, encoding in zip(locations, encodings):
            quality = FaceUtils.get_face_quality(image, location)
            results.append({
                "box": location,
                "encoding": encoding,
                "quality": quality
            })
        
        return results
    
    def register_face(self, user_id: int, image: np.ndarray, db: Session) -> Dict:
        """
        注册用户人脸
        
        Args:
            user_id: 用户ID
            image: RGB图像
            db: 数据库会话
        
        Returns:
            {"success": bool, "quality": str, "message": str}
        """
        # 检测人脸
        faces = self.detect_faces(image)
        
        if len(faces) == 0:
            return {"success": False, "quality": None, "message": "未检测到人脸"}
        
        if len(faces) > 1:
            return {"success": False, "quality": None, "message": "检测到多张人脸，请确保只有一人"}
        
        face = faces[0]
        
        if face["quality"] == "poor":
            return {"success": False, "quality": "poor", "message": "人脸质量较差，请调整光线或位置"}
        
        # 获取用户
        user = db.query(User).filter(User.id == user_id).first()
        if not user:
            return {"success": False, "quality": None, "message": "用户不存在"}
        
        # 保存特征向量
        encoding_filename = f"{user.employee_id}.npy"
        encoding_path = settings.ENCODINGS_DIR / encoding_filename
        np.save(encoding_path, face["encoding"])
        
        # 保存原图
        image_filename = f"{user.employee_id}.jpg"
        image_path = settings.IMAGES_DIR / image_filename
        import cv2
        cv2.imwrite(str(image_path), cv2.cvtColor(image, cv2.COLOR_RGB2BGR))
        
        # 更新数据库
        user.face_encoding_path = str(encoding_path)
        user.face_image_path = str(image_path)
        db.commit()
        
        # 重新加载特征
        self.reload_faces(db)
        
        return {
            "success": True,
            "quality": face["quality"],
            "message": "人脸注册成功"
        }
    
    def recognize_face(self, encoding: np.ndarray) -> Tuple[Optional[Dict], float]:
        """
        1:N 人脸比对
        
        Args:
            encoding: 待识别的人脸特征向量
        
        Returns:
            (user_info, confidence) - 未匹配时 user_info 为 None
        """
        if not self.known_encodings:
            return None, 0.0
        
        # 计算距离
        distances = face_recognition.face_distance(self.known_encodings, encoding)
        
        # 找最小距离
        min_idx = np.argmin(distances)
        min_distance = distances[min_idx]
        
        # 转换为置信度 (0-1)
        confidence = 1 - min_distance
        
        # 判断是否匹配
        if confidence >= self.threshold:
            return self.known_users[min_idx], float(confidence)
        else:
            return None, float(confidence)
    
    def verify_user(self, image: np.ndarray) -> Dict:
        """
        完整验证流程（不含活体检测）
        
        Args:
            image: RGB图像
        
        Returns:
            {
                "success": bool,
                "user": dict | None,
                "confidence": float,
                "reason": str  # 失败原因
            }
        """
        # 检测人脸
        faces = self.detect_faces(image)
        
        if len(faces) == 0:
            return {"success": False, "user": None, "confidence": 0, "reason": "no_face_detected"}
        
        if len(faces) > 1:
            return {"success": False, "user": None, "confidence": 0, "reason": "multiple_faces"}
        
        face = faces[0]
        
        if face["quality"] == "poor":
            return {"success": False, "user": None, "confidence": 0, "reason": "poor_quality"}
        
        # 识别
        user, confidence = self.recognize_face(face["encoding"])
        
        if user is None:
            return {"success": False, "user": None, "confidence": confidence, "reason": "face_not_recognized"}
        
        return {
            "success": True,
            "user": user,
            "confidence": confidence,
            "reason": None
        }


# 全局单例
face_service = FaceService()
```

### 3.4 更新用户管理 API (app/api/users.py)

在 `register_user` 函数中添加人脸处理：

```python
from app.services.face_service import face_service
from app.utils.face_utils import FaceUtils

@router.post("", response_model=UserRegisterResponse, summary="注册新用户")
def register_user(
    employee_id: str = Form(..., description="工号"),
    name: str = Form(..., description="姓名"),
    department: Optional[str] = Form(None, description="部门"),
    face_image: Optional[UploadFile] = File(None, description="人脸照片"),
    db: Session = Depends(get_db)
):
    """注册新用户"""
    # 检查工号
    exists = db.query(User).filter(User.employee_id == employee_id).first()
    if exists:
        raise HTTPException(status_code=400, detail="工号已存在")
    
    # 创建用户
    user = User(employee_id=employee_id, name=name, department=department, status=1)
    db.add(user)
    db.commit()
    db.refresh(user)
    
    # 处理人脸
    face_detected = False
    face_quality = None
    
    if face_image:
        # 读取图像
        image_bytes = face_image.file.read()
        image = FaceUtils.load_image_from_bytes(image_bytes)
        
        # 注册人脸
        result = face_service.register_face(user.id, image, db)
        face_detected = result["success"]
        face_quality = result.get("quality")
        
        if not result["success"]:
            # 人脸注册失败，但用户已创建
            pass  # 可选：删除用户或保留
    
    db.refresh(user)
    response = UserRegisterResponse.model_validate(user)
    response.face_detected = face_detected
    response.face_quality = face_quality
    
    return response
```

## 七、性能优化

1. **模型选择**: HOG 模型更快但精度稍低，CNN 更准但慢（后续可配置）
2. **批量处理**: 多人脸场景可并行处理
3. **缓存策略**: 已加载的特征保持在内存中
4. **GPU加速**: dlib 支持 CUDA，可编译 GPU 版本

## 八、注意事项

1. **dlib 编译**: 首次安装可能需要较长时间
2. **图像格式**: 确保输入为 RGB 格式
3. **内存占用**: 128维特征向量占用很少，可支持数万人脸
4. **并发安全**: 使用全局单例，需注意并发访问
