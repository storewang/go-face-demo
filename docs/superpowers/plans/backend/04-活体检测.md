> 模块: LivenessService - 眨眼/点头检测
> 优先级: P1

> 预计时间: 0.5天

## 一、目标

实现基于动作的活体检测，防止照片、视频等攻击。

## 二、技术方案

### 2.1 眨眼检测 (EAR 算法)

**Eye Aspect Ratio (EAR)** - 眼睛纵横比

```
EAR = (|p2-p6| + |p3-p5|) / (2 * |p1-p4|)

    p1 ─────── p4
    │          │
    p2        p5
    │          │
    p3 ─────── p6
```

- 睁眼时 EAR ≈ 0.2-0.4
- 闭眼时 EAR ≈ 0.1-0.2
- 检测 EAR 从高到低再到高 = 一次眨眼

### 2.2 点头检测 (头部姿态)

通过面部关键点估计头部俯仰角 (pitch)：
- 抬头：pitch > 10°
- 低头：pitch < -10°
- 检测连续帧中的角度变化

## 三、代码实现

### 3.1 活体检测服务 (app/services/liveness_service.py)

```python
import cv2
import dlib
import numpy as np
from typing import List, Dict, Tuple
from scipy.spatial import distance as dist

class LivenessService:
    """活体检测服务"""
    
    def __init__(self, required_frames: int = 5):
        self.required_frames = required_frames
        
        # EAR 阈值
        self.ear_threshold = 0.2
        self.ear_consecutive_frames = 2
        
        # 头部姿态阈值
        self.pitch_threshold = 15  # 度
        
        # 加载 dlib 关键点检测器
        self.detector = dlib.get_frontal_face_detector()
        self.predictor = dlib.shape_predictor("models/shape_predictor_68_face_landmarks.dat")
        
        # 面部关键点索引 (68点模型)
        self.LEFT_EYE_INDICES = list(range(36, 42))
        self.RIGHT_EYE_INDICES = list(range(42, 48))
    
    def calculate_ear(self, eye_landmarks: np.ndarray) -> float:
        """
        计算眼睛纵横比 (Eye Aspect Ratio)
        
        Args:
            eye_landmarks: 6个眼睛关键点 (6, 2)
        
        Returns:
            EAR 值
        """
        # 垂直距离
        v1 = dist.euclidean(eye_landmarks[1], eye_landmarks[5])
        v2 = dist.euclidean(eye_landmarks[2], eye_landmarks[4])
        
        # 水平距离
        h = dist.euclidean(eye_landmarks[0], eye_landmarks[3])
        
        # EAR
        ear = (v1 + v2) / (2.0 * h)
        return ear
    
    def get_landmarks(self, image: np.ndarray, face_location: Tuple) -> np.ndarray:
        """
        获取面部关键点
        
        Args:
            image: 灰度或RGB图像
            face_location: (top, right, bottom, left)
        
        Returns:
            (68, 2) 关键点数组
        """
        top, right, bottom, left = face_location
        rect = dlib.rectangle(left, top, right, bottom)
        
        # 转为灰度图
        if len(image.shape) == 3:
            gray = cv2.cvtColor(image, cv2.COLOR_RGB2GRAY)
        else:
            gray = image
        
        shape = self.predictor(gray, rect)
        landmarks = np.array([[p.x, p.y] for p in shape.parts()])
        
        return landmarks
    
    def estimate_head_pose(self, landmarks: np.ndarray) -> Dict[str, float]:
        """
        估计头部姿态 (简化版 - 仅俯仰角)
        
        Args:
            landmarks: (68, 2) 关键点
        
        Returns:
            {"pitch": float, "yaw": float, "roll": float}  # 单位：度
        """
        # 使用鼻尖和下巴估算俯仰角
        nose = landmarks[30]  # 鼻尖
        chin = landmarks[8]   # 下巴
        forehead = landmarks[27]  # 鼻梁顶部
        
        # 简化计算：基于鼻尖到下巴的垂直距离比例
        face_height = chin[1] - forehead[1]
        nose_position = (nose[1] - forehead[1]) / face_height if face_height > 0 else 0.5
        
        # 标准位置约为 0.5，偏离越大表示俯仰越大
        pitch = (nose_position - 0.5) * 60  # 粗略估算，单位度
        
        return {"pitch": pitch, "yaw": 0, "roll": 0}
    
    def check_blink(self, frames: List[np.ndarray], face_locations: List[Tuple]) -> bool:
        """
        检测眨眼动作
        
        Args:
            frames: 连续帧列表 (RGB)
            face_locations: 对应的人脸位置列表
        
        Returns:
            是否检测到眨眼
        """
        if len(frames) < 3:
            return False
        
        ear_values = []
        
        for frame, location in zip(frames, face_locations):
            landmarks = self.get_landmarks(frame, location)
            
            # 左眼 EAR
            left_eye = landmarks[self.LEFT_EYE_INDICES]
            left_ear = self.calculate_ear(left_eye)
            
            # 右眼 EAR
            right_eye = landmarks[self.RIGHT_EYE_INDICES]
            right_ear = self.calculate_ear(right_eye)
            
            # 平均 EAR
            avg_ear = (left_ear + right_ear) / 2.0
            ear_values.append(avg_ear)
        
        # 检测眨眼模式：高 -> 低 -> 高
        blink_detected = False
        consecutive_low = 0
        
        for i, ear in enumerate(ear_values):
            if ear < self.ear_threshold:
                consecutive_low += 1
            else:
                if consecutive_low >= 1 and consecutive_low <= 3:
                    # 从低恢复到高，算一次眨眼
                    blink_detected = True
                    break
                consecutive_low = 0
        
        return blink_detected
    
    def check_nod(self, frames: List[np.ndarray], face_locations: List[Tuple]) -> bool:
        """
        检测点头动作
        
        Args:
            frames: 连续帧列表
            face_locations: 对应的人脸位置
        
        Returns:
            是否检测到点头
        """
        if len(frames) < 3:
            return False
        
        pitch_values = []
        
        for frame, location in zip(frames, face_locations):
            landmarks = self.get_landmarks(frame, location)
            pose = self.estimate_head_pose(landmarks)
            pitch_values.append(pose["pitch"])
        
        # 检测点头模式：正常 -> 低头 -> 正常 或 正常 -> 抬头 -> 正常
        if len(pitch_values) < 3:
            return False
        
        # 找最大和最小角度
        max_pitch = max(pitch_values)
        min_pitch = min(pitch_values)
        
        # 检查是否有足够的角度变化
        if abs(max_pitch - min_pitch) > self.pitch_threshold:
            return True
        
        return False
    
    def check_liveness(self, frames: List[np.ndarray], face_locations: List[Tuple]) -> Dict:
        """
        综合活体检测
        
        Args:
            frames: 连续帧列表 (3-5帧)
            face_locations: 对应的人脸位置
        
        Returns:
            {
                "passed": bool,
                "blink_detected": bool,
                "nod_detected": bool,
                "message": str
            }
        """
        if len(frames) < 3:
            return {
                "passed": False,
                "blink_detected": False,
                "nod_detected": False,
                "message": "帧数不足"
            }
        
        # 眨眼检测
        blink = self.check_blink(frames, face_locations)
        
        # 点头检测
        nod = self.check_nod(frames, face_locations)
        
        # 通过条件：眨眼 或 点头
        passed = blink or nod
        
        message = "活体检测通过" if passed else "请眨眼或点头"
        
        return {
            "passed": passed,
            "blink_detected": blink,
            "nod_detected": nod,
            "message": message
        }


# 全局单例
liveness_service = None

def get_liveness_service():
    """获取活体检测服务单例"""
    global liveness_service
    if liveness_service is None:
        liveness_service = LivenessService()
    return liveness_service
```

1. **更准确的姿态估计**: 使用 solvePnP 计算精确 3D 姿态

## 九、注意事项

1. **模型文件**: 约 99MB，需提前下载
2. **光照影响**: 低光环境可能影响检测准确性
3. **眼镜干扰**: 戴眼镜可能影响眨眼检测
4. **性能**: 每帧检测约 50-100ms，不适合实时高帧率场景
