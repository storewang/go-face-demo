### 3.4 Composable 测试 (tests/composables/useCamera.test.ts)

```typescript
import { describe, it, expect, vi, beforeEach, afterEach } from 'vitest'
import { useCamera } from '@/composables/useCamera'

// Mock navigator.mediaDevices
Object.defineProperty(global.navigator, 'mediaDevices', {
  value: {
    getUserMedia: vi.fn()
  }
})

describe('useCamera', () => {
  let camera: ReturnType<typeof useCamera>
  
  beforeEach(() => {
    camera = useCamera()
  })
  
  afterEach(() => {
    camera.stop()
  })
  
  it('should start camera', async () => {
    const mockStream = {
      getTracks: () => [{ stop: vi.fn() }]
    }
    
    vi.mocked(navigator.mediaDevices.getUserMedia).mockResolvedValueOnce(mockStream as any)
    
    await camera.start()
    
    expect(camera.isStreaming.value).toBe(true)
    expect(camera.stream.value).toBe(mockStream)
  })
  
  it('should stop camera', () => {
    const mockTrack = { stop: vi.fn() }
    camera.stream.value = {
      getTracks: () => [mockTrack]
    } as any
    camera.isStreaming.value = true
    
    camera.stop()
    
    expect(mockTrack.stop).toHaveBeenCalled()
    expect(camera.isStreaming.value).toBe(false)
  })
})
```

### 5.3 图片压缩

```typescript
// utils/image.ts
export function compressImage(
  file: File,
  maxWidth: number = 640,
  quality: number = 0.8
): Promise<Blob> {
  return new Promise((resolve) => {
    const reader = new FileReader()
    reader.onload = (e) => {
      const img = new Image()
      img.onload = () => {
        const canvas = document.createElement('canvas')
        const ctx = canvas.getContext('2d')!
        
        // 计算缩放比例
        const scale = Math.min(1, maxWidth / img.width)
        canvas.width = img.width * scale
        canvas.height = img.height * scale
        
        ctx.drawImage(img, 0, 0, canvas.width, canvas.height)
        
        canvas.toBlob((blob) => {
          resolve(blob!)
        }, 'image/jpeg', quality)
      }
      img.src = e.target!.result as string
    }
    reader.readAsDataURL(file)
  })
}
```

### 5.5 防抖和节流

```typescript
// utils/debounce.ts
export function debounce<T extends (...args: any[]) => any>(
  fn: T,
  delay: number
): (...args: Parameters<T>) => void {
  let timer: number | null = null
  
  return function(this: any, ...args: Parameters<T>) {
    if (timer) clearTimeout(timer)
    
    timer = window.setTimeout(() => {
      fn.apply(this, args)
    }, delay)
  }
}

export function throttle<T extends (...args: any[]) => any>(
  fn: T,
  delay: number
): (...args: Parameters<T>) => void {
  let last = 0
  
  return function(this: any, ...args: Parameters<T>) {
    const now = Date.now()
    
    if (now - last >= delay) {
      fn.apply(this, args)
      last = now
    }
  }
}
```

### 6.2 Nginx 配置

```nginx
server {
    listen 80;
    server_name your-domain.com;
    
    root /var/www/face-scan/dist;
    index index.html;
    
    # 前端路由
    location / {
        try_files $uri $uri/ /index.html;
    }
    
    # API 代理
    location /api {
        proxy_pass http://localhost:8000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
    }
    
    # WebSocket 代理
    location /ws {
        proxy_pass http://localhost:8000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
    }
    
    # 静态资源缓存
    location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg)$ {
        expires 1y;
        add_header Cache-Control "public, immutable";
    }
}
```

### 6.3 Docker 部署

**Dockerfile**
```dockerfile
# 构建阶段
FROM node:18-alpine as builder

WORKDIR /app

COPY package*.json ./
RUN npm ci

COPY . .
RUN npm run build

# 生产阶段
FROM nginx:alpine

COPY --from=builder /app/dist /usr/share/nginx/html
COPY nginx.conf /etc/nginx/conf.d/default.conf

EXPOSE 80

CMD ["nginx", "-g", "daemon off;"]
```

**docker-compose.yml**
```yaml
version: '3.8'

services:

      - backend
```
