```vue
<template>
  <div class="scan-page">
    <!-- 标题 -->
    <div class="header">
      <h2>人脸识别门禁</h2>
      <el-tag :type="connectionStatus">{{ connectionText }}</el-tag>
    </div>
    
    <!-- 主内容区 -->
    <div class="main-content">

      <div class="camera-section">

          ></canvas>
          
          <!-- 状态提示 -->
          <div v-if="statusMessage" class="status-overlay">

          </div>
        </div>
        
        <!-- 控制按钮 -->
        <div class="controls">
          <el-button
            v-if="!isStreaming"
            type="primary"

<script setup lang="ts">
import { ref, computed, onMounted, onBeforeUnmount } from 'vue'
import { ElMessage } from 'element-plus'
import {
  VideoCamera,
  VideoPause,
  UserFilled,
  CircleCheckFilled,
  WarningFilled,
  Loading
} from '@element-plus/icons-vue'
import { useWebSocket } from '@/composables/useWebSocket'

// Refs
const videoRef = ref<HTMLVideoElement | null>(null)
const overlayRef = ref<HTMLCanvasElement | null>(null)

// 状态
const isStreaming = ref(false)
const isStarting = ref(false)
const statusMessage = ref('')
const statusIcon = ref(Loading)
const lastResult = ref<any>(null)
const showDoorOpen = ref(false)
const scanHistory = ref<any[]>([])

// WebSocket
const {
  isConnected,
  connect,
  disconnect,
  send,
  onMessage
} = useWebSocket()

// 流和帧
let stream: MediaStream | null = null
let frameInterval: number | null = null

// 连接状态
const connectionStatus = computed(() => 
  isConnected.value ? 'success' : 'danger'
)

const connectionText = computed(() => 
  isConnected.value ? '已连接' : '未连接'
)

// 结果样式
const resultClass = computed(() => 
  lastResult.value?.success ? 'success' : 'failed'
)

const resultIcon = computed(() => 
  lastResult.value?.success ? CircleCheckFilled : WarningFilled
)

const resultIconColor = computed(() => 
  lastResult.value?.success ? '#67C23A' : '#F56C6C'
)

const resultTitle = computed(() => 
  lastResult.value?.success ? '识别成功' : '识别失败'
)

const confidenceColor = computed(() => {
  if (!lastResult.value) return '#409EFF'
  const conf = lastResult.value.confidence
  if (conf >= 0.8) return '#67C23A'
  if (conf >= 0.6) return '#E6A23C'
  return '#F56C6C'
})

const userAvatar = computed(() => {
  // TODO: 返回用户头像 URL
  return ''
})

// 开始扫描
async function startScanning() {
  isStarting.value = true
  statusMessage.value = '正在启动摄像头...'
  
  try {
    // 1. 启动摄像头
    stream = await navigator.mediaDevices.getUserMedia({
      video: { width: 640, height: 480, facingMode: 'user' },
      audio: false
    })
    
    if (videoRef.value) {
      videoRef.value.srcObject = stream
      await videoRef.value.play()
    }
    
    // 2. 连接 WebSocket
    await connect()
    
    // 3. 监听消息
    onMessage(handleWebSocketMessage)
    
    // 4. 开始发送帧
    isStreaming.value = true
    statusMessage.value = '请对准摄像头...'
    
    startFrameSending()
    
    ElMessage.success('识别已启动')
  } catch (error: any) {
    console.error('Failed to start scanning:', error)
    ElMessage.error('启动失败: ' + error.message)
    statusMessage.value = '启动失败'
  } finally {
    isStarting.value = false
  }
}

// 停止扫描
function stopScanning() {
  // 停止发送帧
  if (frameInterval) {
    clearInterval(frameInterval)
    frameInterval = null
  }
  
  // 停止摄像头
  if (stream) {
    stream.getTracks().forEach(track => track.stop())
    stream = null
  }
  
  // 断开 WebSocket
  disconnect()
  
  isStreaming.value = false
  statusMessage.value = ''
  lastResult.value = null
  
  ElMessage.info('识别已停止')
}

// 开始发送帧
function startFrameSending() {
  frameInterval = window.setInterval(() => {
    if (!videoRef.value || !isConnected.value) return
    
    const frame = captureFrame()
    if (frame) {
      send({
        type: 'frame',
        data: frame
      })
    }
  }, 200)  // 每 200ms 发送一帧
}

// 捕获帧
function captureFrame(): string | null {
  if (!videoRef.value) return null
  
  const canvas = document.createElement('canvas')
  canvas.width = 640
  canvas.height = 480
  
  const ctx = canvas.getContext('2d')
  if (!ctx) return null
  
  ctx.drawImage(videoRef.value, 0, 0, 640, 480)
  
  return canvas.toDataURL('image/jpeg', 0.8).split(',')[1]
}

// 处理 WebSocket 消息
function handleWebSocketMessage(data: any) {
  if (data.type === 'status') {
    // 状态更新
    statusMessage.value = data.data.message
    statusIcon.value = Loading
    
  } else if (data.type === 'result') {
    // 识别结果
    lastResult.value = data.data
    
    if (data.data.success) {
      statusMessage.value = ''
      showDoorOpen.value = true
      
      // 添加到历史
      scanHistory.value.unshift({
        id: Date.now(),
        name: data.data.user.name,
        action: data.data.action_type === 'CHECK_IN' ? '上班打卡' : '下班打卡',
        time: formatTime(new Date()),
        success: true
      })
      
      // 3秒后隐藏开门动画
      setTimeout(() => {
        showDoorOpen.value = false
      }, 3000)
      
    } else {
      statusMessage.value = data.data.message || '识别失败'
      statusIcon.value = WarningFilled
    }
    
    // 暂停发送帧（冷却时间）
    if (frameInterval) {
      clearInterval(frameInterval)
      setTimeout(() => {
        if (isStreaming.value) {
          startFrameSending()
          lastResult.value = null
        }
      }, 3000)
    }
  }
}

// 重试
function retry() {
  lastResult.value = null
  statusMessage.value = '请对准摄像头...'
}

// 格式化时间
function formatTime(date: Date): string {
  return date.toLocaleTimeString('zh-CN', {
    hour: '2-digit',
    minute: '2-digit',
    second: '2-digit'
  })
}

// 组件卸载
onBeforeUnmount(() => {
  stopScanning()
})
</script>
```

## 四、识别流程

```
用户对准摄像头
       ↓
发送视频帧 (200ms/帧)
       ↓
后端检测人脸
       ↓
  ┌────┴────┐
  │ 未检测到 │ → 返回 "no_face"
  └────┬────┘
       ↓
累积 3-5 帧
       ↓
活体检测
       ↓
  ┌────┴────┐
  │ 未通过  │ → 返回 "liveness_failed"
  └────┬────┘
       ↓
人脸识别
       ↓
  ┌────┴────┐
  │ 未匹配  │ → 返回 "face_not_recognized"
  └────┬────┘
       ↓
识别成功
       ↓
记录考勤
       ↓
返回用户信息 + "门已开启"
       ↓
冷却 3 秒
       ↓
重置状态，等待下一次
```
